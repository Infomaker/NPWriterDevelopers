import {Component, FontAwesomeIcon} from 'substance'
import AddToByline from './AddToByline'
import NilUUID from '../../utils/NilUUID'

/**
 * @typedef Component.BylineComponent.Props
 * @property {Node} node - The node
 * @property {Boolean} bylineSearch - If search should be enabled
 * @property {string} isolatedNodeState
 */

/**
 * A byline component
 * @memberof Component
 *
 * @param {Component.BylineComponent.Props} props
 * @property {Component.BylineComponent.Props} props
 */
class BylineComponent extends Component {

    render($$) {
        const el = $$('div')
        const authorList = $$('ul')
            .addClass('dialog-image-authorlist')
            .attr('contenteditable', false)

        this.props.node.authors.forEach((item) => {
            const authorItem = this._renderAuthor($$, item)
            if (authorItem) {
                authorList.append(authorItem)
            }
        })

        if (['selected', 'focused'].includes(this.props.isolatedNodeState)) {
            authorList.append($$('a')
                .addClass('add-author-link')
                .on('click', this._openAddToByline)
                .append(`+ ${this.getLabel('Add to image byline')}`))
        }

        el.append($$('div')
            .attr('contenteditable', false)
            .addClass('x-im-image-authors')
            .append(authorList)
        )

        return el
    }

    removeAuthor(author) {
        const refId = (NilUUID.isNilUUID(author.uuid)) ? author.name : author.uuid
        const authors = this.props.node.authors

        delete this.refs[`item-${refId}`]

        for (let n = 0; n < authors.length; n++) {
            if (!NilUUID.isNilUUID(author.uuid) && authors[n].uuid === author.uuid) {
                authors.splice(n, 1)
                break
            }
            else if (NilUUID.isNilUUID(author.uuid) && authors[n].name === author.name) {
                authors.splice(n, 1)
                break
            }
        }

        this.props.node.setAuthors(authors)
    }

    _openAddToByline() {
        this.context.api.ui.showDialog(
            AddToByline,
            {
                node: this.props.node,
                bylineSearch: this.props.bylineSearch,
                addAuthor: (author) => {
                    this.props.node.addAuthor(author)
                }
            },
            {
                title: this.getLabel('Add to image byline'),
                global: true,
                primary: this.getLabel('Close'),
                secondary: false
            }
        )
    }

    /**
     * @param $$
     * @param author
     * @returns {VirtualElement}
     * @private
     */
    _renderAuthor($$, author) {
        const refId = (NilUUID.isNilUUID(author.uuid)) ? author.name : author.uuid
        const avatarEl = this._renderAvatar($$, author, refId)

        return $$('li').append(
            $$('div').addClass('author-element').append([
                avatarEl,
                $$('div').append([
                    $$('strong').append(author.name)
                ]),
                $$('span').append(
                    $$('a').append(
                        $$(FontAwesomeIcon, {icon: 'fa-times-circle'})
                    )
                        .attr('title', this.getLabel('Remove'))
                        .on('click', () => {
                            this.removeAuthor(author)
                        })
                )
            ]).ref(`container-${refId}`)
        ).ref(`item-${refId}`);

    }

    /**
     * @param $$
     * @param author
     * @param refId
     * @returns {*}
     * @private
     */
    _renderAvatar($$, author, refId) {
        const Avatar = this.context.api.ui.getComponent('avatar')
        const source = {
            avatarSource: 'gravatar',
            avatarId: author.email
        }

        let twitterHandle
        if(author.links && Avatar._getLinkForType(author.links.link, 'x-im/social+twitter')) {
            if (author.isLoaded && author.links && author.links.link) {
                const twitterLink = Avatar._getLinkForType(author.links.link, 'x-im/social+twitter')
                const twitterURL = Avatar._getTwitterUrlFromAuthorLink(twitterLink)
                if (twitterURL) {
                    twitterHandle = Avatar._getTwitterHandleFromTwitterUrl(twitterURL)
                    source.avatarSource = 'twitter'
                    source.avatarId = twitterHandle
                }
            }
        }

        return $$(Avatar, source).ref(`avatar-${refId}`)
    }
}

export default BylineComponent
