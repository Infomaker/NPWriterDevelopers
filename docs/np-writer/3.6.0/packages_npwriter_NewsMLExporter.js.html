    import {Fragmenter, XMLExporter} from 'substance'

class NewsMLExporter extends XMLExporter {

    constructor(...args) {
        super(...args)
    }

    removeElementIfExists(textEditElement, existingChildren) {
        existingChildren.forEach((child) => {
            if (textEditElement.el.tagName === child.nodeName && textEditElement.el.getAttribute('type') === child.getAttribute('type')) {
                child.remove()
            }
        })
    }

    addHeaderGroup(doc, newsItem) {

        const idfHeaderGroup = newsItem.querySelector('idf group[type="header"]')
        if (!idfHeaderGroup) {
            return
        }

        const textEditComponents = this.context.api.configurator.getTextEditComponents()

        let headerElements = []
        textEditComponents.forEach(textEditComponent => {
            const node = doc.get(textEditComponent.nodeType)
            if (node) {
                const convertedTextEdit = this.convertNode(node)
                headerElements.push(convertedTextEdit.childNodes)
            }
        })

        if (headerElements && headerElements.length > 0) {
            headerElements.forEach((elements) => {
                elements.forEach(element => {
                    this.removeElementIfExists(element, idfHeaderGroup.childNodes)
                    idfHeaderGroup.appendChild(element.el)
                })
            })
        }
    }

    addBodyGroup(doc, newsItem, parent) {
        const serializer = new XMLSerializer()
        const parser = new DOMParser()
        const ownerDocument = parent.ownerDocument

        // Get the first group with type body in IDF section
        const groupNode = newsItem.querySelector('idf group[type="body"]')
        if (!groupNode) {
            throw new Error('IDF node not found!')
        }

        // Export article body with substance convert container function
        // FIXME: Should use createElementNS, but currently this creates extra xmlns attribute.
        // FIXME: Not sure how this is handled in transformers...
        //
        // const nsURI = parent.namespaceURI
        // const bodyGroup = ownerDocument.createElementNS(nsURI, 'group')
        const bodyGroup = ownerDocument.createElement('group')
        bodyGroup.setAttribute('type', 'body')
        for (var node of this.convertContainer(doc.get('body'))) {
            bodyGroup.appendChild(node.el)
        }

        // Clean from control codes
        let cleanGroupNode
        try {
            cleanGroupNode = parser.parseFromString(
                removeControlCodes(serializer.serializeToString(bodyGroup)),
                'text/xml'
            ).firstChild
        }
        catch (err) {
            throw new Error('Failed cleaning control codes: ' + err.message)
        }

        parent.replaceChild(cleanGroupNode, groupNode)
    }

    addTeaser(newsItem, groupContainer) {
        const teaserTypes = this.context.api.getConfigValue('se.infomaker.ximteaser', 'types', false)
        const hasMultiTeaserSupport = teaserTypes !== false

        if(hasMultiTeaserSupport) {
            const queryForMetaTypes = teaserTypes.map(({type}) => `contentMeta > metadata > object[type="${type}"]`).join(',')
            const teaserContainer = newsItem.querySelector('object[type="x-im/teasercontainer"]')
            const teaserMetaElements = newsItem.querySelectorAll(queryForMetaTypes)
            const metadata = newsItem.querySelector('contentMeta > metadata')

            // Extract x-im/*-teaser objects and move it to its correct position if it exists
            teaserMetaElements.forEach((teaserElement) => metadata.removeChild(teaserElement))

            // Add fresh teaser elements to metadata
            if(teaserContainer !== null) {
                [...teaserContainer.children].forEach((newTeaserElement) => metadata.appendChild(newTeaserElement))

                // Remove teasercontainer from xml
                teaserContainer.parentElement.removeChild(teaserContainer)
            }
        } else {
            // Backwards compatibility for teaser plugin

            // Extract x-im/teaser object and move it to its correct position if it exists
            const oldTeaser = newsItem.querySelector('contentMeta > metadata > object[type="x-im/teaser"]');
            const metadata = newsItem.querySelector('contentMeta > metadata');
            if (oldTeaser) {
                metadata.removeChild(oldTeaser);
            }

            const newTeaser = groupContainer.querySelector('object[type="x-im/teaser"]');

            if (newTeaser) {
                newTeaser.parentElement.removeChild(newTeaser);
                metadata.appendChild(newTeaser);
            }
        }
    }


    exportDocument(doc, newsItemArticle) {
        this.state.doc = doc
        const groupContainer = newsItemArticle.querySelector('idf')

        this.addHeaderGroup(doc, newsItemArticle)
        this.addBodyGroup(doc, newsItemArticle, groupContainer)
        this.addTeaser(newsItemArticle, groupContainer)

        const serializer = new XMLSerializer()
        try {
            return removeEmptyNamespaces(
                removeControlCodes(
                    serializer.serializeToString(newsItemArticle.documentElement)
                )
            )
        }
        catch (err) {
            throw new Error('Failed cleaning control codes in article: ' + err.message)
        }
    }

    /**
     * @deprecated
     */
    convert(doc, options, newsItem) {
        console.info("convert method is deprecated, use exportDocument")
        this.state.doc = doc;
        return removeControlCodes(newsItem.documentElement.outerHTML);
    }

    // TODO Remove when/if substance has fixed the double encoding problem
    /**
     * This code is copied from DOMExporter.js in substance. This is because
     * removing the encodeXMLEntities() invocation fixed the double encoding problem (WRIT-255).
     */
    _annotatedText(text, annotations) {
        var self = this

        var annotator = new Fragmenter()
        annotator.onText = function (context, text) {
            // context.children.push(encodeXMLEntities(text))
            context.children.push(text)
        }
        annotator.onEnter = function (fragment) {
            var anno = fragment.node
            return {
                annotation: anno,
                children: []
            }
        }
        annotator.onExit = function (fragment, context, parentContext) {
            var anno = context.annotation
            var converter = self.getNodeConverter(anno)
            if (!converter) {
                converter = self.getDefaultPropertyAnnotationConverter()
            }
            var el
            // Text annotations should not have the same start and end tag, a.k.a being an empty tag WRIT-433
            if (anno.start.equals(anno.end)) {
                el = ""
            } else {
                if (converter.tagName) {
                    el = this.$$(converter.tagName)
                } else {
                    el = this.$$('span')
                }
                el.attr(this.config.idAttribute, anno.id)
                el.append(context.children)
                if (converter.export) {
                    el = converter.export(anno, el, self) || el
                }
            }
            parentContext.children.push(el)
        }.bind(this)
        var wrapper = { children: [] }
        annotator.start(wrapper, text, annotations)
        return wrapper.children
    }

}

export default NewsMLExporter


/**
 * Removes for XML illegal control codes from text (range from 0x00 - 0x1F with the exception of
 * TAB, CR and LF). See http://www.w3.org/TR/xml/#charsets
 * @param {string} serializedXML Serialized xml to process
 * @return {string} Serialized xml without control codes
 */
function removeControlCodes(serializedXML) {
    if (serializedXML === undefined) {
        return serializedXML
    }

    const regex = new RegExp("[\x00-\x08\x0b\x0c\x0e-\x1f]|(\&nbsp\;{1})", "g")
    if (regex.exec(serializedXML) !== null) {
        console.warn("Removing illegal XML character in content")
        return serializedXML.replace(regex, "")
    }

    return serializedXML
}

/**
 * Removes empty xmlns attributes created because the writer and substance
 * dont' create elements with correct namespace.
 *
 * HACK: This should not be necessary when the writer and substance handle namespaces correctly
 *
 * @param {string} serializedXML Serialized xml
 * @return {string} Serialized xml without empty namespaces
 */
function removeEmptyNamespaces(serializedXML) {
    if (serializedXML === undefined) {
        return serializedXML
    }

    const regex = new RegExp(" xmlns=\"\"", "g")
    if (regex.exec(serializedXML) !== null) {
        console.warn("Removing illegal XML character in content")
        return serializedXML.replace(regex, "")
    }

    return serializedXML
}
